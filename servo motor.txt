#include <Servo.h>

Servo servoBase;

void setup() {
   servoBase.attach(A1);//Pin a utilizar para servo
   servoBase.write(0);  //asigno 0 al servo motor
}

void loop() {

  for(int i=0; i<=180; i=i+10)
  {
   servoBase.write(i);
   delay(2000);
  }
 }
